// generate + encapsulate/decapsulate (ML-KEM)
kp = await crypto.subtle.generateKey("ML-KEM-768", true,
    ["encapsulateBits", "encapsulateKey", "decapsulateBits", "decapsulateKey"]);
({ sharedKey, ciphertext } = await crypto.subtle.encapsulateBits("ML-KEM-768", kp.publicKey));
ss = await crypto.subtle.decapsulateBits("ML-KEM-768", kp.privateKey, ciphertext);
new Uint8Array(sharedKey) + "" === new Uint8Array(ss) + ""   // true

// straight to a usable native AES key
({ sharedKey, ciphertext } = await crypto.subtle.encapsulateKey("ML-KEM-768", kp.publicKey,
    { name: "AES-GCM", length: 256 }, false, ["encrypt", "decrypt"]));

// sign / verify (ML-DSA, optional context)
dsa = await crypto.subtle.generateKey("ML-DSA-65", true, ["sign", "verify"]);
sig = await crypto.subtle.sign("ML-DSA-65", dsa.privateKey, new TextEncoder().encode("hi"));
await crypto.subtle.verify("ML-DSA-65", dsa.publicKey, sig, new TextEncoder().encode("hi"));

// export in any format: "jwk" | "pkcs8" | "spki" | "raw-seed" | "raw-public"
await crypto.subtle.exportKey("jwk", dsa.privateKey);

// classical ECDH in the same KEM shape (subtlepq/dhkem extension, RFC 9180);
// migrating to post-quantum = swap the name for "ML-KEM-768"
dh = await crypto.subtle.generateKey("DHKEM-X25519-HKDF-SHA256", false,
    ["encapsulateKey", "decapsulateKey"]);
({ sharedKey, ciphertext } = await crypto.subtle.encapsulateKey("DHKEM-X25519-HKDF-SHA256",
    dh.publicKey, { name: "AES-GCM", length: 256 }, false, ["encrypt"]));

// the module itself is exposed too (ponyfill + install/uninstall/supports)
subtlepq.supports("encapsulateBits", "ML-KEM-1024");
